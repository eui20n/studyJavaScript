# 다양한 정의 방법
```
  // 정의 => 그냥 정의하는 거랑 사실상 같음
  변수 = function(인자1, 인자2, ...){
    실행문;
    return 반환값;
  }
  
  // 호출 => 정의한 변수명으로 호출하면 됨
  
  // 정의와 호출을 한번에 하기 => 정의한 함수명이 없어서 익명함수라고 부름
  (function (인자1, 인자2, ...){
    실행문;
    return 반환값;
  })();
```
* 자바 스크립트는 함수형 언어로 함수를 잘 알아야함
* 익명함수를 잘 알아야함
